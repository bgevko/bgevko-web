## Strings
```python
my_string.upper()
my_string.upper(): # Makes string uppercase.
my_string.lower(): # Makes string lowercase.
my_string.isalpha(): # Returns True if string is alphabetic.
my_string.isalnum(): # Returns True if string is alhpanumeric.

# ASCII Value of a string
ord(my_string) Returns the ascii value of a string.

# Reverse a string
my_string[::-1]

# Split a string a list of characters
list(my_string)

# Split a string based on a delimiter
my_string.split(" ")
```
## Lists
```python
# Enumerate
for count, item in enumerate(iterable, start_index):
	print(count)
	print(item)

# Note, start_index only affects where count starts. All items are still iterated over.

# Sort
list.sort(key=func, reverse=True)

# List Slicing
sublist = list[start:stop]
sublist = list[:stop] # From beginning to specified index
sublist = list[start:] # From specified index to stop
sublist = list[start:stop:step] # From start to stop, stepping by step
sublist = list[::step] # From beginning to end, stepping by step
sublist = list[start::step] # From start to end, stepping by step
sublist = list[:] # Copy of list
sublist = list[-1] # Last item in list
sublist = list[-2] # Second to last item in list
sublist = list[-2:] # Last two items in list
sublist = list[:-2] # Everything except last two items
sublist = list[::-1] # Reversed list
```

## Dictionaries
```python
# Remove elements
my_dict.clear() # Removes all items from dictionary
my_dict.pop(key, default) # Removes specified key and returns its value, or default if key not in dictionary
my_dict.popitem() # Removes and returns the last inserted key-value pair

# Copy elements
other_dict = my_dict.copy() # Returns a copy of my_dict
other_dict = my_dict.fromkeys(keys, value) # New dictionary from specified keys and value

# Get elements
my_dict.get(key, default) # Returns value of key, or default if key not in dictionary
my_dict.setdefault(key, default) # Returns value of key. If the value doesn't exit, inserts the key with the specified value
my_dict.items() # Returns the dictionary's key-value pairs as a list of tuples 
my_dict.keys() # Returns the dictionary's keys as a list
my_dict.values() # Returns the dictionary's values as a list

# Update elements
my_dict.update(other_dict) # Updates my_dict with the key-value pairs from other_dict
```
## Sets
```python
# Set Size
len(set)

# Add
set.add(item)

# Remove
set.remove(item) # Raises KeyError if item not in set

# Discard
set.discard(item) # Does not raise KeyError if item not in set

# Pop
set.pop() # Removes and returns an arbitrary element from set

# Clear
set.clear() # Removes all elements from set

# Union
set.union(set1, set2, set3) # Returns a new set with all items from all sets

# Intersection
set.intersection(set1, set2, set3) # Returns a new set with items that are in all sets

# Difference
my_set.difference(set2) # Returns a new set with elements in my_set but not in set2

# Symmetric Difference
my_set.symmetric_difference(set2) # Returns a new set with elements in either my_set or set2 but not both

# Update
my_set.update(set2) # Adds all elements of set2 to my_set

# Intersection Update
my_set.intersection_update(set2) # Removes all elements of my_set that are not in set2

# Difference Update
my_set.difference_update(set2) # Removes all elements of set2 from my_set

# Symmetric Difference Update
my_set.symmetric_difference_update(set2) # Adds all elements in set2 that are not in my_set, and removes all elements in my_set that are not in set2
my_set.symmetric_difference_update(set2) # Adds all elements in set2 that are not in my_set, and removes all elements in my_set that are not in set2

# Is Subset
my_set.issubset(set2) # Returns True if all elements of my_set are in set2

# Is Superset
my_set.issuperset(set2) # Returns True if all elements of set2 are in my_set

# Is Disjoint
my_set.isdisjoint(set2) # Returns True if my_set has no elements in common with set2
```

## Comprehensions
```python
# List Comprehension
newlist = [x for x in range(10)]
newlist = [x for x in range(10) if x < 5]
newlist = [x.upper() for x in fruits]
newlist = [x if x != "banana" else "orange" for x in fruits]

# Dict Comprehension
newdict = {key: value for vars in iterable}
newdict = {x: x**2 for x in range(10)}
newdict = {x: x**2 for x in range(10) if x%2 == 0}
newdict = {fruit: 'yellow' if fruit == 'banana' else 'not yellow' for fruit in fruits}
newdict = {value: key for key, value in original_dict.items()} # Invert dict
```
